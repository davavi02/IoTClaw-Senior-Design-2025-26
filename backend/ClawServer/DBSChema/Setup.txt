CREATE DATABASE clawser;
CREATE USER 'clawser' IDENTIFIED BY 'reallyhardpassword1!';

//Google Users
  CREATE TABLE GoogleUser( 
  UID INT AUTO_INCREMENT PRIMARY KEY,
  GID VARCHAR(255) NOT NULL UNIQUE,
  Email VARCHAR(255),
  Name VARCHAR(255), 
  Pic TEXT,
  CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

//Machines
CREATE TABLE Machines(
  UID INT AUTO_INCREMENT PRIMARY KEY,
  Name VARCHAR(255) NOT NULL UNIQUE,
  PassHash VARCHAR(255),
  Description TEXT,
  CoinCost INT NOT NULL DEFAULT 1,
  CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

//TokenShop
CREATE TABLE TokenShop(
  UID INT AUTO_INCREMENT PRIMARY KEY,
  NumTokens INT,
  Price DECIMAL(9, 2),
  CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

//UserCoins
CREATE TABLE CoinTotals(
  UID INT PRIMARY KEY,
  Coins INT NOT NULL DEFAULT 0,
  FOREIGN KEY (UID) REFERENCES GoogleUser(UID) ON DELETE CASCADE
);

//ADDRESS Im not gonna think to hard on this..
CREATE TABLE UserAddress(
  UID INT PRIMARY KEY,
  Name VARCHAR(255) NOT NULL,
  Street VARCHAR(255) NOT NULL,
  City VARCHAR(255) NOT NULL,
  Zipcode VARCHAR(255) NOT NULL,
  State VARCHAR(255) NOT NULL,
  FOREIGN KEY (UID) REFERENCES GoogleUser(UID) ON DELETE CASCADE
);

//Prize History — placeholder (use UserPrize for app Prize Tracking)
// CREATE TABLE PrizeHistory( ... );

CREATE TABLE Prize (
  PID INT AUTO_INCREMENT PRIMARY KEY,
  Name VARCHAR(255) NOT NULL,
  Description TEXT,
  PicUrl TEXT
);

// Shipped / physical prizes per user (Prize Tracking screen + GET /api/prizes)
CREATE TABLE UserPrize (
  UniqueId INT AUTO_INCREMENT PRIMARY KEY,
  PID INT NOT NULL,
  UID INT NOT NULL,
  WonAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  IsShipped TINYINT(1) NOT NULL DEFAULT 0,
  IsDelivered TINYINT(1) NOT NULL DEFAULT 0,
  FOREIGN KEY (UID) REFERENCES GoogleUser(UID) ON DELETE CASCADE,
  FOREIGN KEY (PID) REFERENCES Prize(PID) ON DELETE CASCADE
);

// User-submitted bug reports / feedback from the app (Report Error screen)
CREATE TABLE ErrorReport (
  RID INT AUTO_INCREMENT PRIMARY KEY,
  UID INT NOT NULL,
  Subject VARCHAR(200) NOT NULL,
  Message TEXT NOT NULL,
  CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (UID) REFERENCES GoogleUser(UID) ON DELETE CASCADE
);
